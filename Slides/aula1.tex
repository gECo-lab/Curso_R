	\section{O papel do R na Econometria}
	
	%------------------------------------------------
	\begin{frame}{O papel do R na Econometria}
		
		A Econometria busca transformar modelos teóricos em evidências empíricas.
		
		\vspace{0.4cm}
		
		\begin{center}
			\textbf{Teoria econômica + Dados + Métodos estatísticos + Computação}
		\end{center}
		
		\vspace{0.4cm}
		
		O R entra como ferramenta para organizar dados, realizar cálculos, produzir gráficos e estimar modelos.
		
	\end{frame}
	
	%------------------------------------------------
	\begin{frame}{Do modelo teórico ao resultado empírico}
		
		Exemplo simples:
		
		\[
		y = \beta_0 + \beta_1x + \varepsilon
		\]
		
		\vspace{0.3cm}
		
		\begin{itemize}
			\item $y$: variável dependente;
			\item $x$: variável explicativa;
			\item $\beta_1$: efeito estimado de $x$ sobre $y$;
			\item $\varepsilon$: termo de erro.
		\end{itemize}
		
	\end{frame}
	
	%------------------------------------------------
	\begin{frame}{O R como ferramenta acadêmica}
		
		\begin{itemize}
			\item Gratuito e de código aberto;
			\item Muito utilizado em pesquisa econômica;
			\item Forte integração com estatística e econometria;
			\item Permite análises reproduzíveis;
			\item Possui grande comunidade e muitos pacotes.
		\end{itemize}
		
	\end{frame}
	
	%================================================
	\section{O que é o R}
	
	%------------------------------------------------
	\begin{frame}{O que é o R?}
		
		O R é uma linguagem de programação e um ambiente estatístico utilizado para:
		
		\begin{itemize}
			\item manipulação de dados;
			\item análise estatística;
			\item visualização de dados;
			\item estimação de modelos;
			\item produção de relatórios e pesquisas reproduzíveis.
		\end{itemize}
		
	\end{frame}
	
	%------------------------------------------------
	\begin{frame}{R e RStudio}
		
		\begin{block}{R}
			É a linguagem de programação e o ambiente estatístico.
		\end{block}
		
		\begin{block}{RStudio}
			É uma interface que facilita o uso do R.
		\end{block}
		
		\vspace{0.3cm}
		
		\begin{center}
			\textbf{R é o motor. RStudio é o painel de controle.}
		\end{center}
		
	\end{frame}

	\begin{frame}{Conhecendo o RStudio}
		
		O RStudio é organizado em painéis:
		
		\begin{itemize}
			\item \textbf{Script}: onde escrevemos os códigos;
			\item \textbf{Console}: onde os comandos são executados;
			\item \textbf{Environment}: onde aparecem os objetos criados;
			\item \textbf{Files/Plots/Packages}: arquivos, gráficos e pacotes.
		\end{itemize}
		
	\end{frame}
	
	%------------------------------------------------
	\begin{frame}{Script e Console}
		
		\begin{block}{Console}
			Espaço para executar comandos diretamente.
		\end{block}
		
		\begin{block}{Script}
			Arquivo onde escrevemos, salvamos e organizamos os comandos.
		\end{block}
		
		\vspace{0.3cm}
		
		\begin{center}
			\textbf{No curso, o foco será escrever scripts organizados e comentados.}

	\section{Organização e reprodutibilidade}
	
	%------------------------------------------------
	\begin{frame}{Organização de arquivos}
		
		Uma boa análise começa com uma boa organização.
		
		\vspace{0.3cm}
		
		Exemplo de estrutura:
		
		\begin{itemize}
			\item \texttt{dados/}
			\item \texttt{scripts/}
			\item \texttt{resultados/}
			\item \texttt{graficos/}
			\item \texttt{relatorio/}
		\end{itemize}
		
	\end{frame}
	
	%------------------------------------------------
	\begin{frame}{Projetos no RStudio}
		
		Projetos do RStudio ajudam a manter a análise organizada.
		
		\begin{itemize}
			\item Facilitam o uso de caminhos relativos;
			\item Evitam erros com diretórios;
			\item Organizam scripts, dados e resultados;
			\item Melhoram a reprodutibilidade.
		\end{itemize}
		
	\end{frame}
	
	%------------------------------------------------
	\begin{frame}{Boas práticas}
		
		\begin{itemize}
			\item Usar nomes claros para arquivos e objetos;
			\item Comentar partes importantes do código;
			\item Separar dados brutos de dados tratados;
			\item Manter scripts organizados por etapas;
			\item Evitar alterar manualmente bases originais.
		\end{itemize}
		
	\end{frame}
	
	%------------------------------------------------
	\begin{frame}{Erros frequentes}
		
		\begin{itemize}
			\item Não saber onde o arquivo está salvo;
			\item Confundir vírgula e ponto decimal;
			\item Esquecer de instalar ou carregar pacotes;
			\item Usar nomes de objetos diferentes;
			\item Não salvar o script.
		\end{itemize}
		
	\end{frame}
